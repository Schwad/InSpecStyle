def on_block(node)
  return unless inside_file_spec?(node)
  node.descendants.each do |descendant|
    deprecated_file_matcher?(descendant) do |violation|
      add_offense(
        descendant,
        location: offense_range(descendant),
        message: format(
          MSG,
          violation: violation
        )
      )
    end
  end
end
